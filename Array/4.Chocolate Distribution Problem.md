#[Chocolate Distribution Problem](https://www.geeksforgeeks.org/problems/chocolate-distribution-problem3825/1)


``` cpp
class Solution {
  public:
    int findMinDiff(vector<int>& a, int m) {
        // code here
        sort(a.begin(),a.end());
        int n = a.size();
        if(n == 1 && m==1)return 0;
        int minD = a[n-1] - a[0];
        
        for(int i=0;i<=n-m;i++){
            minD = min(minD,a[i+m-1]-a[i]);
        }
        
        return minD;
    }
};
```
